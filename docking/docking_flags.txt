# Input files
-s 5GGS.pdb             # Input PDB with receptor (Y) and drug (X)
-ligand_params_file drug.params  # Ligand parameter file

# Docking protocol
-nstruct 1                     # Number of docking simulations
-spin                            # Allow free rotation of the ligand
-dock_pert 3 8                   # Perturbation: 3 Å translation, 8° rotation
-unboundrot docking_prepack_protocol  # Prepack receptor-ligand structure

# Water placement and optimization
-water:detect                    # Enable automatic placement of water molecules
-water:place_density             # Place waters based on potential hydrogen-bonding density
-water:preoptimize_hbond_geom    # Optimize water hydrogen bond geometry during docking
-water:add_basin_depth 4         # Basin depth for flexible water movement
-water:temperature 0.5           # Water temperature for Monte Carlo sampling

# Scoring and energy function
-score:weights ref15             # Use scoring function that accounts for solvation
-fa_max_dis 9.0                  # Maximum distance for scoring non-bonded interactions

# Output options
-out:file:scorefile score.sc     # Save docking scores
-out:path:pdb 5GGS    # Directory for docked models
